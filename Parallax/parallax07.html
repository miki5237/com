const scroll = () => {
    const newPageYOffset = window.pageYOffset;
    const different = newPageYOffset - pageYOffset;

    // console.log("pageYOffset : " + pageYOffset);
    // console.log("newPageYOffset : " + newPageYOffset);

    gsap.to(".content-item-img",{duration: 0.3, ease: "Power2.out", skewY: different * 0.2});
    gsap.to(".content-item-desc",{duration: 0.3, ease: "Power2.out", skewY: -different * 0.3});

    pageYOffset = newPageYOffset;
    requestAnimationFrame(scroll);
}
