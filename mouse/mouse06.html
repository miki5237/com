const word = document.querySelector(".word");
const shadow = e => {
    const { x,y } = e;
    const rect = word.getBoundingClientRect();
    // console.log(rect)
    const mouseX = (x - rect.left - rect.width/2) / 20;
    const mouseY = (y - rect.top - rect.height/2) / 20;

word.style.textShadow = `${mouseX}px ${mouseY}px 0px rgba(255,255,255,0.7), 
                        ${-mouseX}px ${mouseY}px 0px rgba(255, 166, 166, 0.7),
                        ${mouseX}px ${-mouseY}px 0px rgba(255, 136, 0, 0.7),
                        ${-mouseX}px ${-mouseY}px 0px rgba(251, 255, 0, 0.7)`;

    /* text-shadow: 10px 10px 0 rgba(255,255,255,0.7),
                10px -10px 0 rgba(255, 166, 166, 0.7),
                -10px 10px 0 rgba(255, 136, 0, 0.7),
                -10px -10px 0 rgba(251, 255, 0, 0.7); */
}
window.addEventListener("mousemove",shadow)
